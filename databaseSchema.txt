drop table Employees , Agencies , EmployeesOfAgencies;

CREATE TABLE Employees (
    EmployeeID INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    LastName varchar(255),
    FirstName varchar(255),
    Email varchar(255) NOT NULL UNIQUE,
	Password varchar(255)
);

CREATE TABLE Agencies (
    AgencyID INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    Name varchar(255),
	Address varchar(255),
	OwnerID INTEGER REFERENCES Employees(EmployeeID)
);

CREATE TABLE EmployeesOfAgencies (
	EmployeeID INTEGER REFERENCES Employees(EmployeeID),
	AgencyID INTEGER REFERENCES Agencies(AgencyID)
);
